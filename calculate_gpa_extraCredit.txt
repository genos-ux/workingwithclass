using System;

int examAssignments = 5;

string[] studentNames = {"Sophia","Andrew","Emma","Logan"};

string currentStudentLetterGrade = "";

int[] studentScores = new int[10];

int[] sophiaScores = new int[] { 90, 86, 87, 98, 100, 94, 90 };
int[] andrewScores = new int[] { 92, 89, 81, 96, 90, 89 };
int[] emmaScores = new int[] { 90, 85, 87, 98, 68, 89, 89, 89 };
int[] loganScores = new int[] { 90, 95, 87, 88, 96, 96 };

int sophiaSum = 0;
int andrewSum = 0;
int emmaSum = 0;
int loganSum = 0;

decimal sophiaScore;
decimal andrewScore;
decimal emmaScore;
decimal loganScore;

Console.WriteLine("Student\t\tExam Score\t\tOverall\tGrade\t\tExtra\tCredit\n");

foreach(string name in studentNames){
    string currentName = name;

    if(currentName == "Sophia")
    {
        studentScores = sophiaScores;

    }
    else if(currentName == "Andrew")
    {
        studentScores = andrewScores;
    }
    else if(currentName == "Emma")
    {
        studentScores = emmaScores;
    }
    else if(currentName == "Logan")
    {
        studentScores = loganScores;
    }

    int gradedAssignments = 0;

    int sumExamScores = 0;

    decimal examScore = 0;

    int sumExtraCredit = 0;

    decimal extraCreditScore = 0;

    decimal overallScore = 0;

    decimal extraCredit = 0;

    decimal pts;

    foreach (int score in studentScores)
    {
        gradedAssignments += 1;

        if(gradedAssignments <= examAssignments)
        {
            sumExamScores += score;
        }

        else
        {
            sumExtraCredit += score;
        }
    }

    extraCreditScore = (decimal) sumExtraCredit;

    extraCredit = (decimal) sumExtraCredit / (studentScores.Length - examAssignments);

    examScore = (decimal)sumExamScores / examAssignments;

    overallScore = ((decimal)sumExamScores + (extraCreditScore / 10)) / examAssignments;

    pts = overallScore - examScore;


    if (examScore >= 97)
        currentStudentLetterGrade = "A+";

    else if (examScore >= 93)
        currentStudentLetterGrade = "A";

    else if (examScore >= 90)
        currentStudentLetterGrade = "A-";

    else if (examScore >= 87)
        currentStudentLetterGrade = "B+";

    else if (examScore >= 83)
        currentStudentLetterGrade = "B";

    else if (examScore >= 80)
        currentStudentLetterGrade = "B-";

    else if (examScore >= 77)
        currentStudentLetterGrade = "C+";

    else if (examScore >= 73)
        currentStudentLetterGrade = "C";

    else if (examScore >= 70)
        currentStudentLetterGrade = "C-";

    else if (examScore >= 67)
        currentStudentLetterGrade = "D+";

    else if (examScore >= 63)
        currentStudentLetterGrade = "D";

    else if (examScore >= 60)
        currentStudentLetterGrade = "D-";

    else
        currentStudentLetterGrade = "F";


    Console.WriteLine($"{currentName}\t\t{examScore}\t\t\t{overallScore}\t{currentStudentLetterGrade}\t\t{extraCredit}\t({pts} pts)");


}

Console.WriteLine("Press enter the keyboard");
Console.ReadLine();
